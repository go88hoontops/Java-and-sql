* 조건문(=선택문)
   -if문, switch~case문

1. if문
  
   1)
    if(조건식){
         조건식이 참인경우 실행될 문장;
    }
   
   2)
   if(조건식){
         조건식이 참인경우 실행될 문장;
   }else{
         조건식이 거짓인경우 실행될 문장;
   }

   3)
   if(조건식1){
          조건식1이 참인경우 실행될 문장;
   }else if(조건식2){
          조건식2가 참인경우 실행될 문장;
   }else{
          위의 조건식을 만족하지 않을때 실행될 문장;
   }


2.  switch~case문

     switch(변수){

	case 값1: 실행될 문장;
		break;
	case 값2: 실행될 문장;
		break; 
	default: 위의 case를 만족하지 않을때
	            실행될 문장;
     }
